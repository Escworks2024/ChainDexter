        <button onclick="analyzeTransaction()">Analyze</button>

        <div class="output" id="output"></div>

    <script>

        async function analyzeTransaction() {

            const etherscanApiKey = 'JFVSFTHMS8A8F8KIU1NY44HTDHBMDMYHG8'; // Replace with your Etherscan API key

            const outputDiv = document.getElementById('output');

            outputDiv.innerHTML = 'Fetching transaction details...';


            const input = document.getElementById('transactionLink').value;

            const match = input.match(/https:\/\/etherscan\.io\/tx\/([a-fA-F0-9]+)/);


            if (!match) {

                outputDiv.innerHTML = 'Invalid link. Please enter a valid Etherscan transaction link.';

                return;

            }


            const transactionHash = match[1];

            const apiUrl = `https://api.etherscan.io/api?module=transaction&action=gettxreceiptstatus&txhash=${transactionHash}&apikey=${etherscanApiKey}`;


            try {

                const response = await fetch(apiUrl);

                const data = await response.json();


                if (data.status === '1' && data.result) {

                    const details = `

                        <p><strong>Transaction Hash:</strong> ${transactionHash}</p>

                        <p><strong>Status:</strong> ${data.result.status === '1' ? 'Success' : 'Fail'}</p>

                        <p><strong>Block Number:</strong> ${data.result.blockNumber}</p>

                        <p><strong>Gas Used:</strong> ${data.result.gasUsed}</p>

                    `;

                    outputDiv.innerHTML = details;

                } else {

                    outputDiv.innerHTML = 'Could not retrieve transaction details. Please check the link.';

                }

            } catch (error) {

                outputDiv.innerHTML = 'An error occurred while fetching transaction details. Please try again later.';

                console.error(error);

            }

        }

    </script>
